def receber_arquivo(file_url):

    requisicao = requests.get(file_url)

    # Salvar o arquivo
    if requisicao.ok:
     with open('arquivo_baixado.xlsx', 'wb') as file:
         file.write(requisicao.content)
        print("Arquivo baixado com sucesso")
